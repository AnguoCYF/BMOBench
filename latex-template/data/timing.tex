\newcommand{\plotfile}[1]{
	\pgfplotstableread{timing.txt}{\table}
	\pgfplotstablegetcolsof{timing.txt}
	\pgfmathtruncatemacro\numberofcols{\pgfplotsretval-1}
	\pgfplotsinvokeforeach{1,...,\numberofcols}{
		\pgfplotstablegetcolumnnamebyindex{##1}\of{\table}\to{\colname}
		\addplot table [y index=##1] {timing.txt}; 
		\addlegendentryexpanded{\texttt{\colname}}
	}
}


\begin{document}
	\begin{tikzpicture}
	
	
	\begin{axis}[xmode = log,
	legend style ={ at={(1.03,1)}, 
		anchor=north west, draw=black, 
		fill=white,align=left},
	cycle list name=my black white
	,ylabel=\texttt{runtime/\textbf{f}-evaluation (seconds)}, xlabel=\texttt{\#\textbf{f}-evaluations}]

\plotfile{timing.txt}
	
	
	\end{axis}
	\end{tikzpicture}
